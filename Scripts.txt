--parent table
CREATE TABLE Members (
    MemberID    VARCHAR(10) PRIMARY KEY NOT NULL,
    FirstName   VARCHAR(20)             NOT NULL,
    LastName    VARCHAR(30)             NOT NULL,
    Email       VARCHAR(30),
    Phone       VARCHAR(15),
    Address     VARCHAR(200)            NOT NULL,
    DateOfBirth DATE                    NOT NULL,
    MembershipDate DATE,
    MembershipLevel VARCHAR(50)
);

CREATE TABLE Campaign (
    CampaignID     VARCHAR(10) PRIMARY KEY NOT NULL,
    CampaignName   VARCHAR(50)             NOT NULL,
    StartDate      DATE                    NOT NULL,
    EndDate        DATE                    NOT NULL,
    Description    VARCHAR(50),
    TargetAudience VARCHAR(10)
);

CREATE TABLE Passengers (
    PassengerID     VARCHAR(10) PRIMARY KEY NOT NULL,
    Name            VARCHAR(50)             NOT NULL,
    PassportNumber  VARCHAR(50)             NOT NULL,
    Nationality     VARCHAR(50)             NOT NULL,
    DateOfBirth     DATE                    NOT NULL,
    ContactNumber   VARCHAR(15)             NOT NULL
);

CREATE TABLE Routes (
    RouteID            VARCHAR(10) PRIMARY KEY NOT NULL,
    DepartureLocation  VARCHAR(50)             NOT NULL,
    ArrivalLocation    VARCHAR(50)             NOT NULL,
    Distance           NUMBER(4)               NOT NULL,
    Duration           NUMBER(2)               NOT NULL
);

CREATE TABLE Parts (
    PartID             VARCHAR(10) PRIMARY KEY NOT NULL,
    Name               VARCHAR(50)             NOT NULL,
    Description        VARCHAR(50),
    Manufacturer       VARCHAR(50)             NOT NULL,
    PartNumber         VARCHAR(20)             NOT NULL
);

CREATE TABLE Airplane (
    AirplaneID         VARCHAR(10) PRIMARY KEY NOT NULL,
    Model              VARCHAR(50)             NOT NULL,
    Capacity           NUMBER(3)               NOT NULL,
    Manufacturer       VARCHAR(100)            NOT NULL,
    YearOfManufacture  NUMBER(4)               NOT NULL
);

CREATE TABLE Items (
    ItemID             VARCHAR(10) PRIMARY KEY NOT NULL,
    Name               VARCHAR(30)             NOT NULL,
    Description        VARCHAR(50),
    Price              NUMBER(5, 2)            NOT NULL,
    StockQuantity      NUMBER(5)               NOT NULL
);

CREATE TABLE Staffs (
    StaffID        VARCHAR(10) PRIMARY KEY NOT NULL,
    FirstName      VARCHAR(20)             NOT NULL,
    LastName       VARCHAR(30)             NOT NULL,
    Position       VARCHAR(20)             NOT NULL,
    Email          VARCHAR(30),
    Phone          VARCHAR(15),
    HireDate       DATE                    NOT NULL,
    Salary         NUMBER(10, 2)           NOT NULL
);

CREATE TABLE InsurancePolicy (
    PolicyID         VARCHAR(10) PRIMARY KEY NOT NULL,
    CoverageAmount   NUMBER(10, 2)           NOT NULL,
    PremiumAmount    NUMBER(10, 2)           NOT NULL,
    EffectiveDate    DATE                    NOT NULL,
    ExpirationDate   DATE                    NOT NULL
);


--child table
CREATE TABLE Promotions (
    PromotionID         VARCHAR(10) PRIMARY KEY NOT NULL,
    CampaignID          VARCHAR(10),
    PromotionCode       VARCHAR(20),
    StartDate           DATE,
    EndDate             DATE,
    DiscountPercentage  NUMBER(3),
    FOREIGN KEY (CampaignID) REFERENCES Campaign(CampaignID)
);


CREATE TABLE Claim (
    ClaimID      VARCHAR(10) PRIMARY KEY NOT NULL,
    PolicyID     VARCHAR(10),
    ClaimDate    DATE                    NOT NULL,
    ClaimAmount  NUMBER(10, 2)           NOT NULL,
    Status       VARCHAR(10)             NOT NULL, --(Pending/Approved/Rejected)
    FOREIGN KEY (PolicyID) REFERENCES InsurancePolicy(PolicyID)
);

CREATE TABLE Purchase (
    PurchaseDate   DATE          NOT NULL,
    ItemID         VARCHAR(10)   NOT NULL,
    PassengerID    VARCHAR(10)   NOT NULL,
    TotalAmount    NUMBER(10, 2),
    Quantity       NUMBER(4)     NOT NULL,
    PRIMARY KEY (PurchaseDate, PassengerID, TotalAmount),
    FOREIGN KEY (ItemID) REFERENCES Items(ItemID),
    FOREIGN KEY (PassengerID) REFERENCES Passengers(PassengerID)
);

CREATE TABLE MaintenanceEvent (
    EventID        VARCHAR(10) PRIMARY KEY NOT NULL,
    AirplaneID     VARCHAR(10),
    PartID         VARCHAR(10),
    EventDate      DATE,
    TotalAmount    NUMBER(10, 2),
    FOREIGN KEY (AirplaneID) REFERENCES Airplane(AirplaneID),
    FOREIGN KEY (PartID) REFERENCES Parts(PartID)
);

CREATE TABLE Flights (
    FlightID            VARCHAR(10) PRIMARY KEY NOT NULL,
    RouteID             VARCHAR(10),
    AirplaneID          VARCHAR(10),
    FlightNumber        VARCHAR(20),
    DepartureDateTime   TIMESTAMP               NOT NULL,
    ArrivalDateTime     TIMESTAMP               NOT NULL,
    FOREIGN KEY (RouteID) REFERENCES Routes(RouteID),
    FOREIGN KEY (AirplaneID) REFERENCES Airplane(AirplaneID)
);

CREATE TABLE Workloads (
    WorkloadID          VARCHAR(10) PRIMARY KEY NOT NULL,
    StaffID             VARCHAR(10)             NOT NULL,
    FlightID            VARCHAR(10)             NOT NULL,
    Role                VARCHAR(50)             NOT NULL,
    AssignedDate        DATE                    NOT NULL,
    WorkStartTime       TIMESTAMP               NOT NULL,
    WorkEndTime	        TIMESTAMP               NOT NULL,
    PilotHours	        NUMBER(5),
    FOREIGN KEY (StaffID) REFERENCES Staffs(StaffID),
    FOREIGN KEY (FlightID) REFERENCES Flights(FlightID)
);

CREATE TABLE Booking (
    BookingID     VARCHAR(10) PRIMARY KEY NOT NULL,
    PromotionID   VARCHAR(10),
    MemberID      VARCHAR(10)             NOT NULL,
    TotalPrice    NUMBER(10, 2)           NOT NULL,
    BookingDate   DATE                    NOT NULL,
    FOREIGN KEY (PromotionID) REFERENCES Promotions(PromotionID),
    FOREIGN KEY (MemberID) REFERENCES Members(MemberID)
);


CREATE TABLE CheckLuggages (
    LuggageID NUMBER(10) PRIMARY KEY NOT NULL,
    CheckLuggageID VARCHAR(10) NOT NULL,
    Weight NUMBER(5, 2),
    Status VARCHAR(10), -- checked-in/lost
    TagNumber VARCHAR(20)
);


CREATE TABLE BookingDetail (
    WayID          NUMBER(1)     NOT NULL, --(1 for 1 way; 2 for return)
    BookingID      VARCHAR(10)   NOT NULL,
    PassengerID    VARCHAR(10)   NOT NULL,
    FlightID       VARCHAR(10)   NOT NULL,
    PolicyID       VARCHAR(10)   NOT NULL,
    CheckLuggageID VARCHAR(10),
    SeatNumber     NUMBER(4)     NOT NULL,
    Class          VARCHAR(10)   NOT NULL,
    Price          NUMBER(10, 2) NOT NULL,
    PRIMARY KEY (WayID, BookingID, PassengerID),
    FOREIGN KEY (BookingID)   REFERENCES Booking(BookingID),
    FOREIGN KEY (PassengerID) REFERENCES Passengers(PassengerID),
    FOREIGN KEY (FlightID)    REFERENCES Flights(FlightID),
    FOREIGN KEY (PolicyID)    REFERENCES InsurancePolicy(PolicyID)
);

--Trigger for check the luggage

CREATE OR REPLACE TRIGGER trg_check_luggage_id
BEFORE INSERT OR UPDATE ON CheckLuggages
FOR EACH ROW
DECLARE
    v_count NUMBER;
BEGIN
    -- Check if the CheckLuggageID exists in the BookingDetail table
    SELECT COUNT(*) INTO v_count
    FROM BookingDetail
    WHERE CheckLuggageID = :NEW.CheckLuggageID;

    -- If the CheckLuggageID does not exist, raise an error
    IF v_count = 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'CheckLuggageID does not exist in BookingDetail table.');
    END IF;
END;
/
